<script>
    // --- DATA STORE ---
    const healthData = {
        "india": { modifier: 0.97 },
        "usa": { modifier: 1.05 },
        "africa": { modifier: 0.98 }
    };

    // --- CALCULATOR LOGIC ---
    function generateReport() {
        const age = parseInt(document.getElementById('babyAge').value);
        const region = document.getElementById('regionSelect').value;
        const resultDiv = document.getElementById('result');

        if (!age || age < 1 || age > 60) { alert("Please enter valid age (1-60 months)."); return; }
        if (region === "") { alert("Please select a region."); return; }

        const data = healthData[region];

        // Calc
        let height = (50 + (age * 1.8)) * data.modifier;
        let weight = (3.3 + (age * 0.4)) * data.modifier;
        if(age > 12) {
            height = (75 + ((age-12) * 0.9)) * data.modifier;
            weight = (9 + ((age-12) * 0.2)) * data.modifier;
        }

        // Brain & Skills
        let brain = "";
        let skills = "";
        
        if (age <= 6) {
            brain = "<strong>Sensory Play:</strong> High contrast cards & skin-to-skin bonding.";
            skills = "Rolling over, reaching for toys, smiling.";
        } else if (age <= 12) {
            brain = "<strong>Cause & Effect:</strong> Shaking toys, Peek-a-boo.";
            skills = "Sitting up, crawling, pincer grasp.";
        } else {
            brain = "<strong>Problem Solving:</strong> Stacking blocks, simple puzzles.";
            skills = "Walking, running, speaking words.";
        }

        document.getElementById('resHeight').innerText = height.toFixed(1);
        document.getElementById('resWeight').innerText = weight.toFixed(1);
        document.getElementById('brainText').innerHTML = brain;
        document.getElementById('skillText').innerHTML = skills;
        
        resultDiv.style.display = "grid";
    }

    // --- CHATBOT LOGIC ---
    function toggleChat() {
        const popup = document.getElementById("chatPopup");
        popup.style.display = (popup.style.display === "flex") ? "none" : "flex";
    }
    function handleKeyPress(e) { if(e.key === "Enter") sendMessage(); }
    
    function sendMessage() {
        const input = document.getElementById("userInput");
        const msg = input.value.trim();
        if(!msg) return;
        
        addMsg(msg, "user-msg");
        input.value = "";
        setTimeout(() => { addMsg(getBotReply(msg.toLowerCase()), "bot-msg"); }, 600);
    }

    function addMsg(txt, cls) {
        const div = document.createElement("div"); div.className = "msg " + cls; div.innerHTML = txt;
        const chatBox = document.getElementById("chatMessages"); chatBox.appendChild(div); chatBox.scrollTop = chatBox.scrollHeight;
    }

    function getBotReply(txt) {
        if(txt.includes("recipe") || txt.includes("food")) return "We have recipes for Apple Puree and Lentil Soup in the Nutrition section!";
        if(txt.includes("formula")) return "Remember: Use water at 70¬∞C to mix formula to kill bacteria.";
        if(txt.includes("safety") || txt.includes("car")) return "Don't leave baby in a car seat for more than 2 hours (The 2-Hour Rule).";
        if(txt.includes("brain")) return "Talk to your baby constantly! It builds language pathways.";
        return "I can help with Recipes, Safety rules, or Growth. Try asking 'What should my baby eat?'";
    }
</script>
